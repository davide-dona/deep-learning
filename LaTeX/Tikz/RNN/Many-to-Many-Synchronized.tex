\begin{tikzpicture}[
    item/.style={rectangle, draw, fill=orange!60, minimum size=6mm},
    >=Stealth
]
    % Nodes
    \foreach \i in {1,2,3,4} {
        \node[item] (h\i) at (\i*1.2, 0) {};
        
        % Input at EVERY step
        \draw[<-] (h\i.south) -- ++(0,-0.5);
        
        % Output at EVERY step
        \draw[->] (h\i.north) -- ++(0,0.5);
    }

    % Horizontal recurrence arrows
    \draw[->] (h1) -- (h2);
    \draw[->] (h2) -- (h3);
    \draw[->] (h3) -- (h4);
\end{tikzpicture}
