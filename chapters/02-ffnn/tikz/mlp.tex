\begin{tikzpicture}[
  node distance=1cm and 2cm,
  input/.style={circle, draw=blue!80!black, fill=blue!10, minimum size=22pt, font=\small},
  hidden/.style={circle, draw=blue!80!black, fill=blue!20, minimum size=22pt, font=\small},
  output/.style={circle, draw=blue!80!black, fill=blue!30, minimum size=22pt, font=\small},
  layer label/.style={font=\small\bfseries, text=blue!70!black, align=center},
  conn/.style={->, thick, blue!40!black, opacity=0.5}
]

  % Input layer
  \foreach \i in {1,...,4} {
    \node[input] (i\i) at (0, -\i*1.2) {$x_{\i}$};
  }

  % Hidden layer 1
  \foreach \j in {1,...,3} {
    \node[hidden] (h1\j) at (2.5, -0.6 - \j*1.2) {};
  }

  % Hidden layer 2
  \foreach \j in {1,...,3} {
    \node[hidden] (h2\j) at (5, -0.6 - \j*1.2) {};
  }

  % Hidden layer 3
  \foreach \j in {1,...,3} {
    \node[hidden] (h3\j) at (7.5, -0.6 - \j*1.2) {};
  }

  % Output layer (centered vertically with the network)
  \foreach \k in {1,2} {
    \node[output] (o\k) at (10, -1.2 - \k*1.2) {$y_{\k}$};
  }

  % ──
  \foreach \i in {1,...,4} {
    \foreach \j in {1,...,3} {
      \draw[conn] (i\i) -- (h1\j);
    }
  }

  % ── Connections: Hidden 1 → Hidden 2 ──
  \foreach \i in {1,...,3} {
    \foreach \j in {1,...,3} {
      \draw[conn] (h1\i) -- (h2\j);
    }
  }

  % ── Connections: Hidden 2 → Hidden 3 ──
  \foreach \i in {1,...,3} {
    \foreach \j in {1,...,3} {
      \draw[conn] (h2\i) -- (h3\j);
    }
  }

  % ── Connections: Hidden 3 → Output ──
  \foreach \i in {1,...,3} {
    \foreach \k in {1,2} {
      \draw[conn] (h3\i) -- (o\k);
    }
  }
